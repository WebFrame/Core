44  char* itoa(int num, char* str, int base)
45  {
46  int i = 0;
47  bool isNegative = false;
48 
49  /* Handle 0 explicitly, otherwise empty string is printed for 0 */
50  if (num == 0)
51  {
52  str[i++] = '0';
53  str[i] = '\0';
54  return str;
55  }
56 
57  // In standard itoa(), negative numbers are handled only with
58  // base 10. Otherwise numbers are considered unsigned.
59  if (num < 0 && base == 10)
60  {
61  isNegative = true;
62  num = -num;
63  }
64 
65  // Process individual digits
66  while (num != 0)
67  {
68  int rem = num % base;
69  str[i++] = (rem > 9)? (rem-10) + 'a' : rem + '0';
70  num = num/base;
71  }
72 
73  // If number is negative, append '-'
74  if (isNegative)
75  str[i++] = '-';
76 
77  str[i] = '\0'; // Append string terminator
78 
79  // Reverse the string
80  reverse(str, i);
81 
82  return str;
