140  static const std::string regexAnyChar = "A-Za-z_%0-9.-";
141  std::vector<std::string> v;
142  std::string format = "^";
143  for (size_t i = 0; i < str.size(); i++)
144  {
145  if (str[i] == '{')
146  {
147  std::string var_type;
148  for (i++; i < str.size() and str[i] != '}'; i++)
149  {
150  if (str[i] == ':')
151  break;
152  var_type += str[i];
153  }
154  if (var_type == "")
155  var_type = "string";
156  if (str[i] == '}')
157  {
158  i -= var_type.size() + 1;
159  var_type = "string";
160  }
161  v.push_back(var_type);
162  format += "(";
163  std::string curr_val_regex;
164  for (i++; i < str.size() and str[i] != '}'; i++)
165  {
166  if (str[i] == ':')
167  break;
168  curr_val_regex += str[i];
169  }
170  if (curr_val_regex == "string" or curr_val_regex == "text")
171  curr_val_regex = "[" + regexAnyChar + "]+";
172  if (curr_val_regex == "char" or curr_val_regex == "symbol")
173  curr_val_regex = "[" + regexAnyChar + "]";
174  if (curr_val_regex == "digit")
175  curr_val_regex = "[0-9]";
176  if (curr_val_regex == "number")
177  curr_val_regex = "[0-9]+";
178  if (curr_val_regex == "path")
179  curr_val_regex = "[" + regexAnyChar + "\\/]+";
180  format += curr_val_regex;
181  format += ")";
182  }
183  else if (str[i] == '/') {
184  format += "\\/";
185  }

188  format += str[i];
189  }
190  }

449  request r;
450  try
451  {
452  const std::size_t capacity = 1024;
453  char data[capacity];
454  int total_recv = 0;
455  int n = 0;
456  do
457  {
458  n = RECV(socket, data, capacity, 0);
459  if (n < 0)
460  {
461  break;
462  }
463 
464  total_recv += n;
465 
466  auto state = r.loadMore(data, n);
467 
468  if (state == LoadingState::LOADED) // Headers are loaded, the buffer is not full and the last char is \0 -> body is filled
469  break;
470  } while (n > 0);
471 
472  //this->logger << "(responder) Read state: " << (int)r.getState() << " " << total_recv << "\n";
473 
474  if (r.getState() != LoadingState::LOADED)
475  throw std::string("Request was not loaded completely and data with size=" + std::to_string(total_recv) + " was sent.");
476 
477  const auto t1 = std::chrono::system_clock::now();
478  response res;
479  res = this->respond(r, r.http);
480  const std::string& response = res.to_string();
481  const size_t responseSize = response.size();
482  int status;
483  status = SEND(socket, response.c_str(), responseSize, 0);
484  if (status == SOCKET_ERROR) {
485  return -1;
486  }
487  this->performancer << r.uri << ": " << timer(t1).count() << "miliseconds\n";
488  }

495  SEND(socket, response.c_str(), responseSize, 0);

501  }
