74 convert_path_to_regex(const std::string& str)
75 {
76 static const std::string regexAnyChar = "A-Za-z_%0-9.-";
77 std::vector<std::string> v;
78 std::string format = "^";
79 for (size_t i = 0; i < str.size(); i++)
80 {
81 if (str[i] == '{')
82 {
83 std::string var_type;
84 for (i++; i < str.size() and str[i] != '}'; i++)
85 {
86 if (str[i] == ':')
87 break;
88 var_type += str[i];
89 }
90 if (var_type == "")
91 var_type = "string";
92 if (str[i] == '}')
93 {
94 i -= var_type.size() + 1;
95 var_type = "string";
96 }
97 v.push_back(var_type);
98 format += "(";
99 std::string curr_val_regex;
100 for (i++; i < str.size() and str[i] != '}'; i++)
101 {
102 if (str[i] == ':')
103 break;
104 curr_val_regex += str[i];
105 }
106 if (curr_val_regex == "string" or curr_val_regex == "text")
107 curr_val_regex = "[" + regexAnyChar + "]+";
108 if (curr_val_regex == "char" or curr_val_regex == "symbol")
109 curr_val_regex = "[" + regexAnyChar + "]";
110 if (curr_val_regex == "digit")
111 curr_val_regex = "[0-9]";
112 if (curr_val_regex == "number")
113 curr_val_regex = "[0-9]+";
114 if (curr_val_regex == "path")
115 curr_val_regex = "[" + regexAnyChar + "\\/]+";
116 format += curr_val_regex;
117 format += ")";
118 }
119 else if (str[i] == '/') {
120 format += "\\/";
121 }
122 else
123 {
124 format += str[i];
125 }
126 }
127 format += "$";
128 return {v, std::regex(format)};

371 int responder(int socket)
372 {
373 request r;
374 try
375 {
376 const std::size_t capacity = 1024;
377 char data[capacity];
378 int total_recv = 0;
379 int n = 0;
380 do
381 {
382 n = RECV(socket, data, capacity, 0);
383 if (n < 0)
384 {
385 break;
386 }
387
388 total_recv += n;
389
390 auto state = r.loadMore(data, n);
391
392 if (state == LoadingState::LOADED) // Headers are loaded, the buffer is not full and the last char is \0 -> body is filled
393 break;
394 } while (n > 0);
395
396 this->logger << "(responder) Read state: " << (int)r.getState() << " " << total_recv << "\n";
397
398 if (r.getState() != LoadingState::LOADED)
399 throw std::string("Request was not loaded completely and data with size=" + std::to_string(total_recv) + " was sent.");
400
401 const auto t1 = std::chrono::high_resolution_clock::now();
402 response res;
403 res = this->respond(r, r.http);
404 const std::string& response = res.to_string();
405 const size_t responseSize = response.size();
406 int status;
407 status = SEND(socket, response.c_str(), responseSize, 0);
408 if(status == SOCKET_ERROR) {
409 return -1;
410 }
411 this->performancer << r.uri << ": " << timer(t1).count() << "\n";
412 }
413 catch (std::exception const& e)
414 {
415 this->errors << "(responder) Responding Exception: " << e.what() << "\n";
416 response res = this->responses.at("500").call(r.http, "", path_vars() += {std::string(e.what()), "string"});
417 const std::string& response = res.to_string();
418 const size_t responseSize = response.size();
419 SEND(socket, response.c_str(), responseSize, 0);
420 return -1;
421 }
422 catch (...)
423 {
424 return -2;
425 }
426 return 0;
427 }

533 webframe& run(const char* PORT, const unsigned int cores, bool limited = false, int requests = -1)
534 {
535 this->port_status.initiate(PORT);
536 std::thread([this](const char* PORT, const unsigned int cores, bool limited, int requests) {
537 #ifdef _WIN32
538 // Initialize Winsock.
539 WSADATA wsaData;
540 this->logger << "(main) Startup called\n";
541 int iResult = WSAStartup(MAKEWORD(2, 2), &wsaData);
542 this->logger << "(main) Startup finished " << iResult << "\n";
543 if (iResult != NO_ERROR) {
544 this->errors << "(main) WSAStartup failed with error: " << iResult << "\n";
545 this->port_status.alert_start(PORT);
546 this->port_status.alert_end(PORT);
547 return;
548 }
549 #endif
550 this->logger << "(main) Startup called\n";
551
552 const unsigned int threads = std::min(cores, limited ? requests : cores);
553 std::shared_ptr<thread_pool> threads_ptr = std::make_shared<thread_pool>(threads);
554
555 this->logger << "(main) Thread pool generated\n";
556
557 int status;
558 struct addrinfo hints, *res;
559 int listener;
560
561 // Before using hint you have to make sure that the data structure is empty
562 memset(&hints, 0, sizeof hints);
563 // Set the attribute for hint
564 hints.ai_family = AF_INET; // We don't care V4 AF_INET or 6 AF_INET6
565 hints.ai_socktype = SOCK_STREAM; // TCP Socket SOCK_DGRAM
566 hints.ai_flags = AI_PASSIVE;
567 hints.ai_protocol = 0; /* Any protocol */
568 hints.ai_canonname = NULL;
569 hints.ai_addr = NULL;
570 hints.ai_next = NULL;
571
572 // Fill the res data structure and make sure that the results make sense.
573 status = getaddrinfo(NULL, PORT, &hints, &res);
574 if (status != 0)
575 {
576 this->errors << "(main) getaddrinfo error: " << gai_strerror(status) << "\n";
577 this->port_status.alert_start(PORT);
578 this->port_status.alert_end(PORT);
579 return;
580 }
581
582 // Create Socket and check if error occured afterwards
583 listener = socket(res->ai_family, res->ai_socktype, res->ai_protocol);
584 if (listener == -1)
585 {
586 this->errors << "(main) socket error: " << gai_strerror(status) << "\n";
587 this->port_status.alert_start(PORT);
588 this->port_status.alert_end(PORT);
589 return;
590 }
591
592 // Bind the socket to the address of my local machine and port number
593 status = bind(listener, res->ai_addr, sizeof(*res->ai_addr)/*res->ai_addrlen*/);
594 if (status < 0)
595 {
596 this->errors << "(main) bind error: " << status << " " << gai_strerror(status) << "\n";
597 this->port_status.alert_start(PORT);
598 this->port_status.alert_end(PORT);
599 return;
600 }
601
602 status = listen(listener, 10);
603 if (status < 0)
604 {
605 this->errors << "(main) listen error: " << gai_strerror(status) << "\n";
606 this->port_status.alert_start(PORT);
607 this->port_status.alert_end(PORT);
608 return;
609 }
610
611 status = nonblock_config(listener);
612 if (status < 0)
613 {
614 this->errors << "(main) nonblocking config error: " << gai_strerror(status) << "\n";
615 this->port_status.alert_start(PORT);
616 this->port_status.alert_end(PORT);
617 return;
618 }
619
620 // Free the res linked list after we are done with it
621 freeaddrinfo(res);
622
623 this->logger << "(main) Listener setup " << listener << "\n";
624 bool started = false;
625
626 while (!limited || requests > 0) {
627
628 // Check if abort was requested
629 if(this->port_status.is_over(PORT)) {
630 break;
631 }
632
633 // Check if thread is available to handle a new request
634 const std::optional<size_t> thread = threads_ptr->get_free_thread();
635 if(!thread)
636 continue;
637
638 // Alert waiting for the first request
639 if (!started) {
640 this->port_status.alert_start(PORT);
641 started = true;
642 }
643
644 // Accept a request
645 int client = -1;
646 client = ACCEPT(listener, NULL, NULL);
647 if (client == -1)
648 {
649 continue;
650 }
651
652 this->logger << "(main) Client found: " << client << "\n";
653 this->logger << "(main) Requestor " << client << " is getting handled\n";
654
655 // Check if the socket is valid
656 {
657 struct timeval selTimeout;
658 selTimeout.tv_sec = 2;
659 selTimeout.tv_usec = 0;
660 fd_set readSet;
661 FD_ZERO(&readSet);
662 FD_SET(client + 1, &readSet);
663 FD_SET(client, &readSet);
664
665 status = SELECT(client + 1, &readSet, nullptr, nullptr, &selTimeout);
666 this->logger << "(main) SELECT status is " << status << "\n";
667 if (status < 0) {
668 this->errors << "(main) INVALID SOCKET: " << client << " was skipped (" << status << ")\n";
669 continue;
670 }
671
672 this->logger << "(main) Requestor " << client << " is still valid\n";
673 }
674
675 this->logger << "(main) " << thread.value() << " thread will handle client " << client << "\n";
676 threads_ptr->get(thread.value())->detach(std::make_shared<std::function<void(int)>>([this, &limited, &requests, PORT](int socket) -> void {
677 this->handler (socket, [this, &limited, &requests, PORT]() {
678 if (!limited) return;
679 requests--;
680 this->logger << "(callback) Requests: " << requests << "\n";
681 });
682 }), client);
683 }
684
685 if(!this->port_status.is_over(PORT)) {
686 this->port_status.alert_end(PORT);
687 }
688 #ifdef _WIN32

690 #endif
