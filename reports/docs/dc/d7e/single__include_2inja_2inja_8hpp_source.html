312 inline SourceLocation get_source_location(std::string_view content, size_t pos) {
313  // Get line and offset position (starts at 1:1)
314  auto sliced = string_view::slice(content, 0, pos);
315  std::size_t last_newline = sliced.rfind("\n");
316 
317  if (last_newline == std::string_view::npos) {
318  return {1, sliced.length() + 1};
319  }
320 
321  // Count newlines
322  size_t count_lines = 0;
323  size_t search_start = 0;
324  while (search_start <= sliced.size()) {
325  search_start = sliced.find("\n", search_start) + 1;
326  if (search_start == 0) {
327  break;
328  }
329  count_lines += 1;
330  }
331 
332  return {count_lines + 1, sliced.length() - last_newline};
333 }

1033  Token scan_body(std::string_view close, Token::Kind closeKind, std::string_view close_trim = std::string_view(), bool trim = false) {
1034  again:
1035  // skip whitespace (except for \n as it might be a close)
1036  if (tok_start >= m_in.size()) {
1037  return make_token(Token::Kind::Eof);
1038  }
1039  const char ch = m_in[tok_start];
1040  if (ch == ' ' || ch == '\t' || ch == '\r') {
1041  tok_start += 1;
1042  goto again;
1043  }
1044 
1045  // check for close
1046  if (!close_trim.empty() && inja::string_view::starts_with(m_in.substr(tok_start), close_trim)) {
1047  state = State::Text;
1048  pos = tok_start + close_trim.size();
1049  const Token tok = make_token(closeKind);
1050  skip_whitespaces_and_newlines();
1051  return tok;
1052  }
1053 
1054  if (inja::string_view::starts_with(m_in.substr(tok_start), close)) {
1055  state = State::Text;
1056  pos = tok_start + close.size();
1057  const Token tok = make_token(closeKind);
1058  if (trim) {
1059  skip_whitespaces_and_first_newline();
1060  }
1061  return tok;
1062  }
1063 
1064  // skip \n
1065  if (ch == '\n') {
1066  tok_start += 1;
1067  goto again;
1068  }
1069 
1070  pos = tok_start + 1;
1071  if (std::isalpha(ch)) {
1072  minus_state = MinusState::Operator;
1073  return scan_id();
1074  }
1075 
1076  const MinusState current_minus_state = minus_state;
1077  if (minus_state == MinusState::Operator) {
1078  minus_state = MinusState::Number;
1079  }
1080 
1081  switch (ch) {
1082  case '+':
1083  return make_token(Token::Kind::Plus);
1084  case '-':
1085  if (current_minus_state == MinusState::Operator) {
1086  return make_token(Token::Kind::Minus);
1087  }
1088  return scan_number();
1089  case '*':
1090  return make_token(Token::Kind::Times);
1091  case '/':
1092  return make_token(Token::Kind::Slash);
1093  case '^':
1094  return make_token(Token::Kind::Power);
1095  case '%':
1096  return make_token(Token::Kind::Percent);
1097  case '.':
1098  return make_token(Token::Kind::Dot);
1099  case ',':
1100  return make_token(Token::Kind::Comma);
1101  case ':':
1102  return make_token(Token::Kind::Colon);
1103  case '(':
1104  return make_token(Token::Kind::LeftParen);
1105  case ')':
1106  minus_state = MinusState::Operator;
1107  return make_token(Token::Kind::RightParen);
1108  case '[':
1109  return make_token(Token::Kind::LeftBracket);
1110  case ']':
1111  minus_state = MinusState::Operator;
1112  return make_token(Token::Kind::RightBracket);
1113  case '{':
1114  return make_token(Token::Kind::LeftBrace);
1115  case '}':
1116  minus_state = MinusState::Operator;
1117  return make_token(Token::Kind::RightBrace);
1118  case '>':
1119  if (pos < m_in.size() && m_in[pos] == '=') {
1120  pos += 1;
1121  return make_token(Token::Kind::GreaterEqual);
1122  }
1123  return make_token(Token::Kind::GreaterThan);
1124  case '<':
1125  if (pos < m_in.size() && m_in[pos] == '=') {
1126  pos += 1;
1127  return make_token(Token::Kind::LessEqual);
1128  }
1129  return make_token(Token::Kind::LessThan);
1130  case '=':
1131  if (pos < m_in.size() && m_in[pos] == '=') {
1132  pos += 1;
1133  return make_token(Token::Kind::Equal);
1134  }
1135  return make_token(Token::Kind::Unknown);
1136  case '!':
1137  if (pos < m_in.size() && m_in[pos] == '=') {
1138  pos += 1;
1139  return make_token(Token::Kind::NotEqual);
1140  }
1141  return make_token(Token::Kind::Unknown);
1142  case '\"':
1143  return scan_string();
1144  case '0':
1145  case '1':
1146  case '2':
1147  case '3':
1148  case '4':
1149  case '5':
1150  case '6':
1151  case '7':
1152  case '8':
1153  case '9':
1154  minus_state = MinusState::Operator;
1155  return scan_number();
1156  case '_':
1157  case '@':
1158  case '$':
1159  minus_state = MinusState::Operator;
1160  return scan_id();
1161  default:
1162  return make_token(Token::Kind::Unknown);
1163  }
1164  }

1166  Token scan_id() {
1167  for (;;) {
1168  if (pos >= m_in.size()) {
1169  break;
1170  }
1171  const char ch = m_in[pos];
1172  if (!std::isalnum(ch) && ch != '.' && ch != '/' && ch != '_' && ch != '-') {
1173  break;
1174  }
1175  pos += 1;
1176  }

1181  for (;;) {
1182  if (pos >= m_in.size()) {
1183  break;
1184  }
1185  const char ch = m_in[pos];
1186  // be very permissive in lexer (we'll catch errors when conversion happens)
1187  if (!(std::isdigit(ch) || ch == '.' || ch == 'e' || ch == 'E' || (ch == '+' && (pos == 0 || m_in[pos-1] == 'e' || m_in[pos-1] == 'E')) || (ch == '-' && (pos == 0 || m_in[pos-1] == 'e' || m_in[pos-1] == 'E')))) {
1188  break;
1189  }
1190  pos += 1;
1191  }
1192  return make_token(Token::Kind::Number);
1193  }

1267  void start(std::string_view input) {
1268  m_in = input;
1269  tok_start = 0;
1270  pos = 0;
1271  state = State::Text;
1272  minus_state = MinusState::Number;
1273 
1274  // Consume byte order mark (BOM) for UTF-8

1280  Token scan() {
1281  tok_start = pos;
1282 
1283  again:
1284  if (tok_start >= m_in.size()) {
1285  return make_token(Token::Kind::Eof);
1286  }
1287 
1288  switch (state) {
1289  default:
1290  case State::Text: {
1291  // fast-scan to first open character
1292  const size_t open_start = m_in.substr(pos).find_first_of(config.open_chars);
1293  if (open_start == std::string_view::npos) {
1294  // didn't find open, return remaining text as text token
1295  pos = m_in.size();
1296  return make_token(Token::Kind::Text);
1297  }
1298  pos += open_start;
1299 
1300  // try to match one of the opening sequences, and get the close
1301  std::string_view open_str = m_in.substr(pos);
1302  bool must_lstrip = false;
1303  if (inja::string_view::starts_with(open_str, config.expression_open)) {
1304  if (inja::string_view::starts_with(open_str, config.expression_open_force_lstrip)) {
1305  state = State::ExpressionStartForceLstrip;
1306  must_lstrip = true;
1307  } else {
1308  state = State::ExpressionStart;
1309  }
1310  } else if (inja::string_view::starts_with(open_str, config.statement_open)) {
1311  if (inja::string_view::starts_with(open_str, config.statement_open_no_lstrip)) {
1312  state = State::StatementStartNoLstrip;
1313  } else if (inja::string_view::starts_with(open_str, config.statement_open_force_lstrip)) {
1314  state = State::StatementStartForceLstrip;
1315  must_lstrip = true;
1316  } else {
1317  state = State::StatementStart;
1318  must_lstrip = config.lstrip_blocks;
1319  }
1320  } else if (inja::string_view::starts_with(open_str, config.comment_open)) {
1321  if (inja::string_view::starts_with(open_str, config.comment_open_force_lstrip)) {
1322  state = State::CommentStartForceLstrip;
1323  must_lstrip = true;
1324  } else {
1325  state = State::CommentStart;
1326  must_lstrip = config.lstrip_blocks;
1327  }
1328  } else if ((pos == 0 || m_in[pos - 1] == '\n') && inja::string_view::starts_with(open_str, config.line_statement)) {
1329  state = State::LineStart;
1330  } else {
1331  pos += 1; // wasn't actually an opening sequence
1332  goto again;
1333  }
1334 
1335  std::string_view text = string_view::slice(m_in, tok_start, pos);
1336  if (must_lstrip) {
1337  text = clear_final_line_if_whitespace(text);
1338  }
1339 
1340  if (text.empty()) {
1341  goto again; // don't generate empty token
1342  }
1343  return Token(Token::Kind::Text, text);
1344  }
1345  case State::ExpressionStart: {
1346  state = State::ExpressionBody;
1347  pos += config.expression_open.size();
1348  return make_token(Token::Kind::ExpressionOpen);
1349  }
1350  case State::ExpressionStartForceLstrip: {
1351  state = State::ExpressionBody;
1352  pos += config.expression_open_force_lstrip.size();
1353  return make_token(Token::Kind::ExpressionOpen);
1354  }
1355  case State::LineStart: {
1356  state = State::LineBody;
1357  pos += config.line_statement.size();
1358  return make_token(Token::Kind::LineStatementOpen);
1359  }
1360  case State::StatementStart: {
1361  state = State::StatementBody;
1362  pos += config.statement_open.size();
1363  return make_token(Token::Kind::StatementOpen);
1364  }
1365  case State::StatementStartNoLstrip: {
1366  state = State::StatementBody;
1367  pos += config.statement_open_no_lstrip.size();
1368  return make_token(Token::Kind::StatementOpen);
1369  }
1370  case State::StatementStartForceLstrip: {
1371  state = State::StatementBody;
1372  pos += config.statement_open_force_lstrip.size();
1373  return make_token(Token::Kind::StatementOpen);
1374  }
1375  case State::CommentStart: {
1376  state = State::CommentBody;
1377  pos += config.comment_open.size();
1378  return make_token(Token::Kind::CommentOpen);
1379  }
1380  case State::CommentStartForceLstrip: {
1381  state = State::CommentBody;
1382  pos += config.comment_open_force_lstrip.size();
1383  return make_token(Token::Kind::CommentOpen);
1384  }
1385  case State::ExpressionBody:
1386  return scan_body(config.expression_close, Token::Kind::ExpressionClose, config.expression_close_force_rstrip);
1387  case State::LineBody:
1388  return scan_body("\n", Token::Kind::LineStatementClose);
1389  case State::StatementBody:
1390  return scan_body(config.statement_close, Token::Kind::StatementClose, config.statement_close_force_rstrip, config.trim_blocks);
1391  case State::CommentBody: {
1392  // fast-scan to comment close
1393  const size_t end = m_in.substr(pos).find(config.comment_close);
1394  if (end == std::string_view::npos) {
1395  pos = m_in.size();
1396  return make_token(Token::Kind::Eof);
1397  }
1398 
1399  // Check for trim pattern
1400  const bool must_rstrip = inja::string_view::starts_with(m_in.substr(pos + end - 1), config.comment_close_force_rstrip);
1401 
1402  // return the entire comment in the close token
1403  state = State::Text;
1404  pos += end + config.comment_close.size();
1405  Token tok = make_token(Token::Kind::CommentClose);
1406 
1407  if (must_rstrip || config.trim_blocks) {
1408  skip_whitespaces_and_first_newline();
1409  }
1410  return tok;
1411  }
1412  }

1552  bool parse_expression(Template& tmpl, Token::Kind closing) {
1553  while (tok.kind != closing && tok.kind != Token::Kind::Eof) {
1554  // Literals
1555  switch (tok.kind) {
1556  case Token::Kind::String: {
1557  if (current_brace_level == 0 && current_bracket_level == 0) {
1558  literal_start = tok.text;
1559  add_literal(tmpl.content.c_str());
1560  }
1561  } break;
1562  case Token::Kind::Number: {
1563  if (current_brace_level == 0 && current_bracket_level == 0) {
1564  literal_start = tok.text;
1565  add_literal(tmpl.content.c_str());
1566  }
1567  } break;
1568  case Token::Kind::LeftBracket: {
1569  if (current_brace_level == 0 && current_bracket_level == 0) {
1570  literal_start = tok.text;
1571  }
1572  current_bracket_level += 1;
1573  } break;
1574  case Token::Kind::LeftBrace: {
1575  if (current_brace_level == 0 && current_bracket_level == 0) {
1576  literal_start = tok.text;
1577  }
1578  current_brace_level += 1;
1579  } break;
1580  case Token::Kind::RightBracket: {
1581  if (current_bracket_level == 0) {
1582  throw_parser_error("unexpected ']'");
1583  }
1584 
1585  current_bracket_level -= 1;
1586  if (current_brace_level == 0 && current_bracket_level == 0) {
1587  add_literal(tmpl.content.c_str());
1588  }
1589  } break;
1590  case Token::Kind::RightBrace: {
1591  if (current_brace_level == 0) {
1592  throw_parser_error("unexpected '}'");
1593  }
1594 
1595  current_brace_level -= 1;
1596  if (current_brace_level == 0 && current_bracket_level == 0) {
1597  add_literal(tmpl.content.c_str());
1598  }
1599  } break;
1600  case Token::Kind::Id: {
1601  get_peek_token();
1602 
1603  // Data Literal
1604  if (tok.text == static_cast<decltype(tok.text)>("true") || tok.text == static_cast<decltype(tok.text)>("false") ||
1605  tok.text == static_cast<decltype(tok.text)>("null")) {
1606  if (current_brace_level == 0 && current_bracket_level == 0) {
1607  literal_start = tok.text;
1608  add_literal(tmpl.content.c_str());
1609  }
1610 
1611  // Operator
1612  } else if (tok.text == "and" || tok.text == "or" || tok.text == "in" || tok.text == "not") {
1613  goto parse_operator;
1614 
1615  // Functions
1616  } else if (peek_tok.kind == Token::Kind::LeftParen) {
1617  operator_stack.emplace(std::make_shared<FunctionNode>(static_cast<std::string>(tok.text), tok.text.data() - tmpl.content.c_str()));
1618  function_stack.emplace(operator_stack.top().get(), current_paren_level);
1619 
1620  // Variables
1621  } else {
1622  arguments.emplace_back(std::make_shared<DataNode>(static_cast<std::string>(tok.text), tok.text.data() - tmpl.content.c_str()));
1623  }
1624 
1625  // Operators
1626  } break;
1627  case Token::Kind::Equal:
1628  case Token::Kind::NotEqual:
1629  case Token::Kind::GreaterThan:
1630  case Token::Kind::GreaterEqual:
1631  case Token::Kind::LessThan:
1632  case Token::Kind::LessEqual:
1633  case Token::Kind::Plus:
1634  case Token::Kind::Minus:
1635  case Token::Kind::Times:
1636  case Token::Kind::Slash:
1637  case Token::Kind::Power:
1638  case Token::Kind::Percent:
1639  case Token::Kind::Dot: {
1640 
1641  parse_operator:
1642  FunctionStorage::Operation operation;
1643  switch (tok.kind) {
1644  case Token::Kind::Id: {
1645  if (tok.text == "and") {
1646  operation = FunctionStorage::Operation::And;
1647  } else if (tok.text == "or") {
1648  operation = FunctionStorage::Operation::Or;
1649  } else if (tok.text == "in") {
1650  operation = FunctionStorage::Operation::In;
1651  } else if (tok.text == "not") {
1652  operation = FunctionStorage::Operation::Not;
1653  } else {
1654  throw_parser_error("unknown operator in parser.");
1655  }
1656  } break;
1657  case Token::Kind::Equal: {
1658  operation = FunctionStorage::Operation::Equal;
1659  } break;
1660  case Token::Kind::NotEqual: {
1661  operation = FunctionStorage::Operation::NotEqual;
1662  } break;
1663  case Token::Kind::GreaterThan: {
1664  operation = FunctionStorage::Operation::Greater;
1665  } break;
1666  case Token::Kind::GreaterEqual: {
1667  operation = FunctionStorage::Operation::GreaterEqual;
1668  } break;
1669  case Token::Kind::LessThan: {
1670  operation = FunctionStorage::Operation::Less;
1671  } break;
1672  case Token::Kind::LessEqual: {
1673  operation = FunctionStorage::Operation::LessEqual;
1674  } break;
1675  case Token::Kind::Plus: {
1676  operation = FunctionStorage::Operation::Add;
1677  } break;
1678  case Token::Kind::Minus: {
1679  operation = FunctionStorage::Operation::Subtract;
1680  } break;
1681  case Token::Kind::Times: {
1682  operation = FunctionStorage::Operation::Multiplication;
1683  } break;
1684  case Token::Kind::Slash: {
1685  operation = FunctionStorage::Operation::Division;
1686  } break;
1687  case Token::Kind::Power: {
1688  operation = FunctionStorage::Operation::Power;
1689  } break;
1690  case Token::Kind::Percent: {
1691  operation = FunctionStorage::Operation::Modulo;
1692  } break;
1693  case Token::Kind::Dot: {
1694  operation = FunctionStorage::Operation::AtId;
1695  } break;
1696  default: {
1697  throw_parser_error("unknown operator in parser.");
1698  }
1699  }
1700  auto function_node = std::make_shared<FunctionNode>(operation, tok.text.data() - tmpl.content.c_str());
1701 
1702  while (!operator_stack.empty() &&
1703  ((operator_stack.top()->precedence > function_node->precedence) ||
1704  (operator_stack.top()->precedence == function_node->precedence && function_node->associativity == FunctionNode::Associativity::Left)) &&
1705  (operator_stack.top()->operation != FunctionStorage::Operation::ParenLeft)) {
1706  add_operator();
1707  }
1708 
1709  operator_stack.emplace(function_node);
1710  } break;
1711  case Token::Kind::Comma: {
1712  if (current_brace_level == 0 && current_bracket_level == 0) {
1713  if (function_stack.empty()) {
1714  throw_parser_error("unexpected ','");
1715  }
1716 
1717  function_stack.top().first->number_args += 1;
1718  }
1719  } break;
1720  case Token::Kind::Colon: {
1721  if (current_brace_level == 0 && current_bracket_level == 0) {
1722  throw_parser_error("unexpected ':'");
1723  }
1724  } break;
1725  case Token::Kind::LeftParen: {
1726  current_paren_level += 1;
1727  operator_stack.emplace(std::make_shared<FunctionNode>(FunctionStorage::Operation::ParenLeft, tok.text.data() - tmpl.content.c_str()));
1728 
1729  get_peek_token();
1730  if (peek_tok.kind == Token::Kind::RightParen) {
1731  if (!function_stack.empty() && function_stack.top().second == current_paren_level - 1) {
1732  function_stack.top().first->number_args = 0;
1733  }
1734  }
1735  } break;
1736  case Token::Kind::RightParen: {
1737  current_paren_level -= 1;
1738  while (!operator_stack.empty() && operator_stack.top()->operation != FunctionStorage::Operation::ParenLeft) {
1739  add_operator();
1740  }
1741 
1742  if (!operator_stack.empty() && operator_stack.top()->operation == FunctionStorage::Operation::ParenLeft) {
1743  operator_stack.pop();
1744  }
1745 
1746  if (!function_stack.empty() && function_stack.top().second == current_paren_level) {
1747  auto func = function_stack.top().first;
1748  auto function_data = function_storage.find_function(func->name, func->number_args);
1749  if (function_data.operation == FunctionStorage::Operation::None) {
1750  throw_parser_error("unknown function " + func->name);
1751  }
1752  func->operation = function_data.operation;
1753  if (function_data.operation == FunctionStorage::Operation::Callback) {
1754  func->callback = function_data.callback;
1755  }
1756 
1757  if (operator_stack.empty()) {
1758  throw_parser_error("internal error at function " + func->name);
1759  }
1760 
1761  add_operator();
1762  function_stack.pop();
1763  }
1764  }
1765  default:
1766  break;
1767  }
1768 
1769  get_next_token();
1770  }
1771 
1772  while (!operator_stack.empty()) {
1773  add_operator();
1774  }
1775 
1776  if (arguments.size() == 1) {
1777  current_expression_list->root = arguments[0];
1778  arguments = {};
1779  } else if (arguments.size() > 1) {
1780  throw_parser_error("malformed expression");
1781  }
1782 
1783  return true;
1784  }

1786  bool parse_statement(Template& tmpl, Token::Kind closing, std::string_view path) {
1787  if (tok.kind != Token::Kind::Id) {
1788  return false;
1789  }
1790 
1791  if (tok.text == static_cast<decltype(tok.text)>("if")) {
1792  get_next_token();
1793 
1794  auto if_statement_node = std::make_shared<IfStatementNode>(current_block, tok.text.data() - tmpl.content.c_str());
1795  current_block->nodes.emplace_back(if_statement_node);
1796  if_statement_stack.emplace(if_statement_node.get());
1797  current_block = &if_statement_node->true_statement;
1798  current_expression_list = &if_statement_node->condition;
1799 
1800  if (!parse_expression(tmpl, closing)) {
1801  return false;
1802  }
1803  } else if (tok.text == static_cast<decltype(tok.text)>("else")) {
1804  if (if_statement_stack.empty()) {
1805  throw_parser_error("else without matching if");
1806  }
1807  auto& if_statement_data = if_statement_stack.top();
1808  get_next_token();
1809 
1810  if_statement_data->has_false_statement = true;
1811  current_block = &if_statement_data->false_statement;
1812 
1813  // Chained else if
1814  if (tok.kind == Token::Kind::Id && tok.text == static_cast<decltype(tok.text)>("if")) {
1815  get_next_token();
1816 
1817  auto if_statement_node = std::make_shared<IfStatementNode>(true, current_block, tok.text.data() - tmpl.content.c_str());
1818  current_block->nodes.emplace_back(if_statement_node);
1819  if_statement_stack.emplace(if_statement_node.get());
1820  current_block = &if_statement_node->true_statement;
1821  current_expression_list = &if_statement_node->condition;
1822 
1823  if (!parse_expression(tmpl, closing)) {
1824  return false;
1825  }
1826  }
1827  } else if (tok.text == static_cast<decltype(tok.text)>("endif")) {
1828  if (if_statement_stack.empty()) {
1829  throw_parser_error("endif without matching if");
1830  }
1831 
1832  // Nested if statements
1833  while (if_statement_stack.top()->is_nested) {
1834  if_statement_stack.pop();
1835  }
1836 
1837  auto& if_statement_data = if_statement_stack.top();
1838  get_next_token();
1839 
1840  current_block = if_statement_data->parent;
1841  if_statement_stack.pop();
1842  } else if (tok.text == static_cast<decltype(tok.text)>("block")) {
1843  get_next_token();
1844 
1845  if (tok.kind != Token::Kind::Id) {
1846  throw_parser_error("expected block name, got '" + tok.describe() + "'");
1847  }
1848 
1849  const std::string block_name = static_cast<std::string>(tok.text);
1850 
1851  auto block_statement_node = std::make_shared<BlockStatementNode>(current_block, block_name, tok.text.data() - tmpl.content.c_str());
1852  current_block->nodes.emplace_back(block_statement_node);
1853  block_statement_stack.emplace(block_statement_node.get());
1854  current_block = &block_statement_node->block;
1855  auto success = tmpl.block_storage.emplace(block_name, block_statement_node);
1856  if (!success.second) {
1857  throw_parser_error("block with the name '" + block_name + "' does already exist");
1858  }
1859 
1860  get_next_token();
1861  } else if (tok.text == static_cast<decltype(tok.text)>("endblock")) {
1862  if (block_statement_stack.empty()) {
1863  throw_parser_error("endblock without matching block");
1864  }
1865 
1866  auto& block_statement_data = block_statement_stack.top();
1867  get_next_token();
1868 
1869  current_block = block_statement_data->parent;
1870  block_statement_stack.pop();
1871  } else if (tok.text == static_cast<decltype(tok.text)>("for")) {
1872  get_next_token();
1873 
1874  // options: for a in arr; for a, b in obj
1875  if (tok.kind != Token::Kind::Id) {
1876  throw_parser_error("expected id, got '" + tok.describe() + "'");
1877  }
1878 
1879  Token value_token = tok;
1880  get_next_token();
1881 
1882  // Object type
1883  std::shared_ptr<ForStatementNode> for_statement_node;
1884  if (tok.kind == Token::Kind::Comma) {
1885  get_next_token();
1886  if (tok.kind != Token::Kind::Id) {
1887  throw_parser_error("expected id, got '" + tok.describe() + "'");
1888  }
1889 
1890  Token key_token = std::move(value_token);
1891  value_token = tok;
1892  get_next_token();
1893 
1894  for_statement_node = std::make_shared<ForObjectStatementNode>(static_cast<std::string>(key_token.text), static_cast<std::string>(value_token.text),
1895  current_block, tok.text.data() - tmpl.content.c_str());
1896 
1897  // Array type
1898  } else {
1899  for_statement_node =
1900  std::make_shared<ForArrayStatementNode>(static_cast<std::string>(value_token.text), current_block, tok.text.data() - tmpl.content.c_str());
1901  }
1902 
1903  current_block->nodes.emplace_back(for_statement_node);
1904  for_statement_stack.emplace(for_statement_node.get());
1905  current_block = &for_statement_node->body;
1906  current_expression_list = &for_statement_node->condition;
1907

1914  return false;
1915  }

1917  if (for_statement_stack.empty()) {
1918  throw_parser_error("endfor without matching for");
1919  }
1920 
1921  auto& for_statement_data = for_statement_stack.top();
1922  get_next_token();
1923 
1924  current_block = for_statement_data->parent;
1925  for_statement_stack.pop();
1926  } else if (tok.text == static_cast<decltype(tok.text)>("include")) {
1927  get_next_token();

1930  add_to_template_storage(path, template_name);
1931 
1932  current_block->nodes.emplace_back(std::make_shared<IncludeStatementNode>(template_name, tok.text.data() - tmpl.content.c_str()));
1933 
1934  get_next_token();
1935  } else if (tok.text == static_cast<decltype(tok.text)>("extends")) {
1936  get_next_token();
1937 
1938  std::string template_name = parse_filename(tok);
1939  add_to_template_storage(path, template_name);
1940 
1941  current_block->nodes.emplace_back(std::make_shared<ExtendsStatementNode>(template_name, tok.text.data() - tmpl.content.c_str()));
1942 
1943  get_next_token();
1944  } else if (tok.text == static_cast<decltype(tok.text)>("set")) {
1945  get_next_token();
1946 
1947  if (tok.kind != Token::Kind::Id) {
1948  throw_parser_error("expected variable name, got '" + tok.describe() + "'");
1949  }
1950 
1951  std::string key = static_cast<std::string>(tok.text);
1952  get_next_token();
1953 
1954  auto set_statement_node = std::make_shared<SetStatementNode>(key, tok.text.data() - tmpl.content.c_str());
1955  current_block->nodes.emplace_back(set_statement_node);
1956  current_expression_list = &set_statement_node->expression;
1957 
1958  if (tok.text != static_cast<decltype(tok.text)>("=")) {
1959  throw_parser_error("expected '=', got '" + tok.describe() + "'");
1960  }
1961  get_next_token();
1962 
1963  if (!parse_expression(tmpl, closing)) {
1964  return false;
1965  }
1966  } else {
1967  return false;
1968  }
1969  return true;

1972  void parse_into(Template& tmpl, std::string_view path) {
1973  lexer.start(tmpl.content);
1974  current_block = &tmpl.root;
1975 
1976  for (;;) {
1977  get_next_token();
1978  switch (tok.kind) {
1979  case Token::Kind::Eof: {
1980  if (!if_statement_stack.empty()) {
1981  throw_parser_error("unmatched if");
1982  }
1983  if (!for_statement_stack.empty()) {
1984  throw_parser_error("unmatched for");
1985  }
1986  }
1987  return;
1988  case Token::Kind::Text: {
1989  current_block->nodes.emplace_back(std::make_shared<TextNode>(tok.text.data() - tmpl.content.c_str(), tok.text.size()));
1990  } break;
1991  case Token::Kind::StatementOpen: {
1992  get_next_token();
1993  if (!parse_statement(tmpl, Token::Kind::StatementClose, path)) {
1994  throw_parser_error("expected statement, got '" + tok.describe() + "'");
1995  }
1996  if (tok.kind != Token::Kind::StatementClose) {
1997  throw_parser_error("expected statement close, got '" + tok.describe() + "'");
1998  }
1999  } break;
2000  case Token::Kind::LineStatementOpen: {
2001  get_next_token();
2002  if (!parse_statement(tmpl, Token::Kind::LineStatementClose, path)) {
2003  throw_parser_error("expected statement, got '" + tok.describe() + "'");
2004  }
2005  if (tok.kind != Token::Kind::LineStatementClose && tok.kind != Token::Kind::Eof) {
2006  throw_parser_error("expected line statement close, got '" + tok.describe() + "'");
2007  }
2008  } break;
2009  case Token::Kind::ExpressionOpen: {
2010  get_next_token();
2011 
2012  auto expression_list_node = std::make_shared<ExpressionListNode>(tok.text.data() - tmpl.content.c_str());
2013  current_block->nodes.emplace_back(expression_list_node);
2014  current_expression_list = expression_list_node.get();
2015 
2016  if (!parse_expression(tmpl, Token::Kind::ExpressionClose)) {
2017  throw_parser_error("expected expression, got '" + tok.describe() + "'");
2018  }
2019 
2020  if (tok.kind != Token::Kind::ExpressionClose) {
2021  throw_parser_error("expected expression close, got '" + tok.describe() + "'");
2022  }
2023  } break;
2024  case Token::Kind::CommentOpen: {
2025  get_next_token();
2026  if (tok.kind != Token::Kind::CommentClose) {
2027  throw_parser_error("expected comment close, got '" + tok.describe() + "'");
2028  }
2029  } break;
2030  default: {
2031  throw_parser_error("unexpected token '" + tok.describe() + "'");
2032  } break;
2033  }
2034  }
2035  }
