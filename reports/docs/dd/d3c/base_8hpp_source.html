45 char* itoa(int num, char* str, int base)
46 {
47 int i = 0;
48 bool isNegative = false;
49
50 /* Handle 0 explicitly, otherwise empty string is printed for 0 */
51 if (num == 0)
52 {
53 str[i++] = '0';
54 str[i] = '\0';
55 return str;
56 }
57
58 // In standard itoa(), negative numbers are handled only with
59 // base 10. Otherwise numbers are considered unsigned.
60 if (num < 0 && base == 10)
61 {
62 isNegative = true;
63 num = -num;
64 }
65
66 // Process individual digits
67 while (num != 0)
68 {
69 int rem = num % base;
70 str[i++] = (rem > 9)? (rem-10) + 'a' : rem + '0';
71 num = num/base;
72 }
73
74 // If number is negative, append '-'
75 if (isNegative)
76 str[i++] = '-';
77
78 str[i] = '\0'; // Append string terminator
79
80 // Reverse the string
81 reverse(str, i);
82
83 return str;
84 }
