558  void parse_into(Template& tmpl, std::string_view path) {
559  lexer.start(tmpl.content);
560  current_block = &tmpl.root;
561 
562  for (;;) {
563  get_next_token();
564  switch (tok.kind) {
565  case Token::Kind::Eof: {
566  if (!if_statement_stack.empty()) {
567  throw_parser_error("unmatched if");
568  }
569  if (!for_statement_stack.empty()) {
570  throw_parser_error("unmatched for");
571  }
572  }
573  return;
574  case Token::Kind::Text: {
575  current_block->nodes.emplace_back(std::make_shared<TextNode>(tok.text.data() - tmpl.content.c_str(), tok.text.size()));
576  } break;
577  case Token::Kind::StatementOpen: {
578  get_next_token();
579  if (!parse_statement(tmpl, Token::Kind::StatementClose, path)) {
580  throw_parser_error("expected statement, got '" + tok.describe() + "'");
581  }
582  if (tok.kind != Token::Kind::StatementClose) {
583  throw_parser_error("expected statement close, got '" + tok.describe() + "'");
584  }
585  } break;
586  case Token::Kind::LineStatementOpen: {
587  get_next_token();
588  if (!parse_statement(tmpl, Token::Kind::LineStatementClose, path)) {
589  throw_parser_error("expected statement, got '" + tok.describe() + "'");
590  }
591  if (tok.kind != Token::Kind::LineStatementClose && tok.kind != Token::Kind::Eof) {
592  throw_parser_error("expected line statement close, got '" + tok.describe() + "'");
593  }
594  } break;
595  case Token::Kind::ExpressionOpen: {
596  get_next_token();
597 
598  auto expression_list_node = std::make_shared<ExpressionListNode>(tok.text.data() - tmpl.content.c_str());
599  current_block->nodes.emplace_back(expression_list_node);
600  current_expression_list = expression_list_node.get();
601 
602  if (!parse_expression(tmpl, Token::Kind::ExpressionClose)) {
603  throw_parser_error("expected expression, got '" + tok.describe() + "'");
604  }
605 
606  if (tok.kind != Token::Kind::ExpressionClose) {
607  throw_parser_error("expected expression close, got '" + tok.describe() + "'");
608  }
609  } break;
610  case Token::Kind::CommentOpen: {
611  get_next_token();
612  if (tok.kind != Token::Kind::CommentClose) {
613  throw_parser_error("expected comment close, got '" + tok.describe() + "'");
614  }
615  } break;
616  default: {
617  throw_parser_error("unexpected token '" + tok.describe() + "'");
618  } break;
619  }
620  }
